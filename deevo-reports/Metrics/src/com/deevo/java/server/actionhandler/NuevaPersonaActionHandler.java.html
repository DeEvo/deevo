25      }

27      @Override
28      public NuevaPersonaResult execute(NuevaPersona action, ExecutionContext context)
29              throws ActionException {
30              Persona persona = new Persona();
31              persona.setPerDni(action.getDni());
32              persona.setPerNom(action.getNombre());
33              persona.setPerPat(action.getAppatern());
34              persona.setPerMat(action.getApmatern());
35              persona.setPerDir(action.getDireccion());
36              persona.setPerCel(action.getCelular());
37              persona.setPerTelf(action.getTelefono());
38              persona.setPerNac(action.getFechanac());
39              persona.setPerEmail(action.getEmail());
40              persona.setPerEstc(action.getEstc());
41              persona.setPerIng(new Date());
42              persona.setSexo(action.getSexo());
43              PersonaDao perdao = new PersonaDao();
44              try{
45              perdao.createPersona(persona);
46              }catch(EntityExistsException a){
47                  throw new ActionException("La Persona ya existe "+a.getMessage());
48              }catch(SQLException a){ 
49                  throw new ActionException("La Persona ya existe ");
50              }catch(Throwable a){
51                  throw new ActionException("Solicitar ayuda al administrador");
52              }
53                  
54              if(action.getFlag_usuario()){
55                  Usuario user = new Usuario();
56                  
57                  user.setUsurCod(action.getNombre().substring(0, 2).toLowerCase() + action.getAppatern().toLowerCase() + action.getApmatern().substring(0, 2).toLowerCase());
58                  user.setPerPass(action.getDni());
59                  user.setPersona(persona);
60                                  
61                  UsuarioDao userdao = new UsuarioDao();
62                  while(userdao.existeUsuario(user)){
63                      Random r = new Random();
64                      user.setUsurCod( action.getNombre().substring(0, 2).toLowerCase() + action.getAppatern().toLowerCase() + action.getApmatern().substring(0, 2).toLowerCase() +  (char)(r.nextInt(26)+97));
65                  }
66                  try {
67                      userdao.createUsuario(user);
68                  } catch (Throwable e) {
69                      // TODO Auto-generated catch block
70                      e.printStackTrace();
71                      throw new ActionException("La Persona ya existe ");
72                  }
73                  return new NuevaPersonaResult("Exito", user.getUsurCod() , user.getPerPass());
74              }           
75          return new NuevaPersonaResult("Exito", null , null);
76      }

81      }
