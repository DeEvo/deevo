1   package com.deevo.java.server.actionhandler;
2   
3   import javax.persistence.EntityExistsException;
4   
5   import com.gwtplatform.dispatch.server.actionhandler.ActionHandler;
6   import com.deevo.java.client.action.NuevoPrivilegio;
7   import com.deevo.java.client.action.NuevoPrivilegioResult;
8   import com.deevo.java.server.model.dao.PrivilegioDao;
9   import com.deevo.java.shared.Privilegio;
10  import com.google.inject.Inject;
11  import com.gwtplatform.dispatch.server.ExecutionContext;
12  import com.gwtplatform.dispatch.shared.ActionException;
13  
14  public class NuevoPrivilegioActionHandler implements
15          ActionHandler<NuevoPrivilegio, NuevoPrivilegioResult> {
16  
17      @Inject
18      public NuevoPrivilegioActionHandler() {
19      }
20  
21      @Override
22      public NuevoPrivilegioResult execute(NuevoPrivilegio action,
23              ExecutionContext context) throws ActionException {
24           Privilegio privilegio = new Privilegio();
25           
26          privilegio.setPriDes(action.getPri_des());
27          privilegio.setPriFun(action.getPri_fun());
28          privilegio.setPriNom(action.getPri_nom());
29          
30          PrivilegioDao privilegiodao = new PrivilegioDao();
31          
32          try {
33              privilegiodao.createPrivilegio(privilegio);
34          }catch(EntityExistsException a){
35              throw new ActionException("La Persona ya existe "+a.getMessage());
36          }catch(Throwable a){
37              throw new ActionException("Solicitar ayuda al administrador");
38          }
39          return new NuevoPrivilegioResult("EXITO");
40      }
41  
42      @Override
43      public void undo(NuevoPrivilegio action, NuevoPrivilegioResult result,
44              ExecutionContext context) throws ActionException {
45      }
46  
47      @Override
48      public Class<NuevoPrivilegio> getActionType() {
49          return NuevoPrivilegio.class;
50      }
51  }
